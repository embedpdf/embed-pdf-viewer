# Pan (Hand Tool)

The Pan plugin provides a "hand tool" mode that allows users to click and drag to scroll the document. This is often preferred over text selection for navigation, especially on touch devices.

## Configuration

You can configure when the pan tool should be active by default using the `pan` config object.

```svelte
<PDFViewer
  config={{
    pan: {
      // 'mobile' | 'always' | 'never'
      defaultMode: 'mobile' 
    }
  }}
/>
```

### Modes

| Mode | Description |
| :--- | :--- |
| `'mobile'` | **Default**. Pan is enabled by default only on touch devices. Desktop users start in text selection mode. |
| `'always'` | Pan is always the default mode. |
| `'never'` | Text selection (or another mode) is always the default. |

### Toggling Mode

```svelte
<script lang="ts">
  import { PDFViewer, type PluginRegistry, type PanPlugin } from '@embedpdf/svelte-pdf-viewer';

  let registry = $state<PluginRegistry | null>(null);

  const handleReady = (r: PluginRegistry) => {
    registry = r;
  };

  const togglePan = () => {
    const panPlugin = registry?.getPlugin<PanPlugin>('pan')?.provides();
    const docPan = panPlugin?.forDocument('my-doc-id');
    docPan?.togglePan();
  };

  const enablePan = () => {
    const panPlugin = registry?.getPlugin<PanPlugin>('pan')?.provides();
    const docPan = panPlugin?.forDocument('my-doc-id');
    docPan?.enablePan();
  };

  const disablePan = () => {
    const panPlugin = registry?.getPlugin<PanPlugin>('pan')?.provides();
    const docPan = panPlugin?.forDocument('my-doc-id');
    docPan?.disablePan();
  };
</script>

<PDFViewer onready={handleReady} config={{ src: '/doc.pdf' }} />
```

### Checking State

```svelte
<script lang="ts">
  const isPanning = () => {
    const panPlugin = registry?.getPlugin<PanPlugin>('pan')?.provides();
    const docPan = panPlugin?.forDocument('my-doc-id');
    return docPan?.isPanMode();
  };
</script>
```

## Listening for Changes

You can listen for changes in the interaction mode to update your UI (e.g., highlighting the active tool).

```svelte
<script lang="ts">
  import { PDFViewer, type PluginRegistry } from '@embedpdf/svelte-pdf-viewer';

  let toolState = $state<'hand' | 'cursor'>('cursor');

  const handleReady = (registry: PluginRegistry) => {
    const panPlugin = registry.getPlugin('pan')?.provides();

    // Subscribe to global changes
    panPlugin?.onPanModeChange(({ documentId, isPanMode }) => {
      console.log(`Document ${documentId} pan mode is now: ${isPanMode}`);
    });

    // Or subscribe to a specific document
    const docPan = panPlugin?.forDocument('my-doc');
    docPan?.onPanModeChange((isPanMode) => {
      toolState = isPanMode ? 'hand' : 'cursor';
    });
  };
</script>

<PDFViewer onready={handleReady} config={{ src: '/doc.pdf' }} />
```
